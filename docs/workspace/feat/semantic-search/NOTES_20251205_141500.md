```json
{
  "mcpServers": {
    "frontmatter": {
      "command": "uvx",
      "args": [
        "--with", "sentence-transformers",
        "frontmatter-mcp",
        "--base-dir", "/path/to/vault"
      ],
      "env": {
        "FRONTMATTER_ENABLE_SEMANTIC": "true"
      }
    }
  }
}
```
